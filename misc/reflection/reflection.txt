Divisor(n):=(
  regional(out,tmp);
  out=[];
  forall(1..(round(sqrt(n))),
    if(mod(n,#)==0,
      tmp=round(n/#);
      out=concat(out,[#,tmp]));
  );
  out=sort(out);
  out;
);
